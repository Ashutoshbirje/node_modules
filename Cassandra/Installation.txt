# COMMAND 
> USE "keyspace_name";
> DESCRIBE KEYSPACES;
> CREATE KEYSPACE "keyspace_name" 
  WITH REPLICATION = {'class' : 'SimpleStrategy', 'replication_factor' : 3};
> CREATE TABLE "keyspace_name".table_name (
  column1 data_type PRIMARY KEY,
  column2 data_type,
  ...
  );
> SELECT * FROM "keyspace_name".table_name;
